/system script set [find name="github-smart-route-list-sync"] source={
    :local url "https://raw.githubusercontent.com/zmin511/smart-route-list/main/smart-route-list.txt"
    :local targetList "smart-route-list"

    :local added 0
    :local updated 0
    :local reenabled 0
    :local cntDisabled 0
    :local cntDisableErrors 0
    :local skipped 0
    :local bad 0
    :local skippedExisting 0

    :log info ("github-smart-route-list-sync: start, list=" . $targetList)

    :log info ("github-smart-route-list-sync: disabling current static entries in " . $targetList)

    :foreach i in=[/ip firewall address-list find where list=$targetList and dynamic=no] do={
        :do {
            :local curDisabled [/ip firewall address-list get $i disabled]
            :if ($curDisabled = no) do={
                /ip firewall address-list set $i disabled=yes
                :set cntDisabled ($cntDisabled + 1)
            }
        } on-error={
            :set cntDisableErrors ($cntDisableErrors + 1)
            :log warning ("github-smart-route-list-sync: failed to disable item " . $i)
        }
    }

    :log info ("github-smart-route-list-sync: disable stage finished, disabled=" . $cntDisabled . ", errors=" . $cntDisableErrors)

    :do {
        :local result [/tool fetch url=$url http-method=get output=user as-value]
        :local data ($result->"data")

        :if ([:len $data] = 0) do={
            :log error "github-smart-route-list-sync: empty response"
            :error "empty response"
        }

        :local pos 0
        :local totalLen [:len $data]

        :while ($pos < $totalLen) do={
            :local eol [:find $data "\n" $pos]
            :if ($eol = nil) do={ :set eol $totalLen }

            :local line [:pick $data $pos $eol]
            :set pos ($eol + 1)

            :if ([:len $line] > 0) do={
                :if ([:pick $line ([:len $line] - 1) [:len $line]] = "\r") do={
                    :set line [:pick $line 0 ([:len $line] - 1)]
                }
            }

            :if ([:len $line] = 0) do={
                :set skipped ($skipped + 1)
                :continue
            }

            :local firstChar [:pick $line 0 1]
            :if (($firstChar = "#") || ($firstChar = ";")) do={
                :set skipped ($skipped + 1)
                :continue
            }

            :local delim [:find $line "#"]
            :if ($delim = nil) do={
                :set bad ($bad + 1)
                :log warning ("github-smart-route-list-sync: bad line without #: " . $line)
                :continue
            }

            :local addr [:pick $line 0 $delim]
            :local commentText [:pick $line ($delim + 1) [:len $line]]

            :while (([:len $addr] > 0) and ([:pick $addr 0 1] = " ")) do={
                :set addr [:pick $addr 1 [:len $addr]]
            }
            :while (([:len $addr] > 0) and ([:pick $addr ([:len $addr] - 1) [:len $addr]] = " ")) do={
                :set addr [:pick $addr 0 ([:len $addr] - 1)]
            }

            :while (([:len $commentText] > 0) and ([:pick $commentText 0 1] = " ")) do={
                :set commentText [:pick $commentText 1 [:len $commentText]]
            }
            :while (([:len $commentText] > 0) and ([:pick $commentText ([:len $commentText] - 1) [:len $commentText]] = " ")) do={
                :set commentText [:pick $commentText 0 ([:len $commentText] - 1)]
            }

            :if (([:len $commentText] > 0) and ([:pick $commentText 0 1] = "\"")) do={
                :set commentText [:pick $commentText 1 [:len $commentText]]
            }
            :if (([:len $commentText] > 0) and ([:pick $commentText ([:len $commentText] - 1) [:len $commentText]] = "\"")) do={
                :set commentText [:pick $commentText 0 ([:len $commentText] - 1)]
            }

            :if (([:len $addr] = 0) || ([:len $commentText] = 0)) do={
                :set bad ($bad + 1)
                :log warning ("github-smart-route-list-sync: bad line after trim: " . $line)
                :continue
            }

            :local staticFound 0
            :local anyFound 0

            :set anyFound [:len [/ip firewall address-list find where list=$targetList and address=$addr]]
            :local existingStatic [/ip firewall address-list find where list=$targetList and address=$addr and dynamic=no]

            :if ([:len $existingStatic] > 0) do={
                :foreach item in=$existingStatic do={
                    :do {
                        /ip firewall address-list set $item comment=$commentText disabled=no
                        :set updated ($updated + 1)
                        :set staticFound 1
                    } on-error={
                        :set bad ($bad + 1)
                        :log warning ("github-smart-route-list-sync: failed to update static item for line: " . $line)
                    }
                }
            }

            :if ($staticFound = 1) do={
                :continue
            }

            :if ($anyFound > 0) do={
                :set skippedExisting ($skippedExisting + 1)
                :log warning ("github-smart-route-list-sync: skipped because entry already exists as dynamic or conflicting item: " . $line)
                :continue
            }

            :do {
                /ip firewall address-list add list=$targetList address=$addr comment=$commentText disabled=no
                :set added ($added + 1)
            } on-error={
                :set bad ($bad + 1)
                :log warning ("github-smart-route-list-sync: failed on line: " . $line)
            }
        }

        :log info ("github-smart-route-list-sync: done, list=" . $targetList . ", disabled=" . $cntDisabled . ", disableErrors=" . $cntDisableErrors . ", added=" . $added . ", updated=" . $updated . ", skippedExisting=" . $skippedExisting . ", skipped=" . $skipped . ", bad=" . $bad)

    } on-error={
        :log error "github-smart-route-list-sync: fetch/parse failed"
    }
}
}
/system scheduler
add name=github-smart-route-list-sync-01h start-time=01:00:00 interval=1d \
    policy=read,write,test,policy,ftp \
    on-event="/system script run github-smart-route-list-sync"
/system script run github-smart-route-list-sync
/ip firewall filter add chain=forward dst-address-list=smart-route-list action=accept
/routing/table
add name=to-vpn fib
/ip firewall mangle
add chain=output dst-address-list=smart-route-list action=mark-routing \
    new-routing-mark=to-vpn passthrough=no comment="router itself -> VPN"
/ip route
add dst-address=0.0.0.0/0 gateway=wireguard-out routing-table=to-vpn
